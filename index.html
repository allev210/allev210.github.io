  const baseUrl = document.getElementById('cloud-region').value;

  const orgIdInput = document.getElementById('org-id').value.trim();

  const errEl = document.getElementById('conn-error');

  const btn = document.getElementById('connect-btn');


  errEl.style.display = 'none';

  if (!token) { showError('API token is required.'); return; }


  btn.disabled = true; btn.textContent = 'Connecting…';

  state.token = token; state.baseUrl = baseUrl;


  try {

    // Auto-detect org if not provided

    let orgId = orgIdInput;

    if (!orgId) {

      const selfResp = await mistFetch('/api/v1/self');

      if (!selfResp.ok) throw new Error(`Auth failed (${selfResp.status}). Check your API token.`);

      const self = await selfResp.json();

      const orgs = Object.keys(self.privileges?.reduce((a, p) => {

        if (p.org_id) a[p.org_id] = true; return a;

      }, {}) || {});

      if (!orgs.length) throw new Error('No organizations found for this token.');

      orgId = orgs[0];

    }

    state.orgId = orgId;


    // Load sites for lookup

    const sitesResp = await mistFetch(`/api/v1/orgs/${orgId}/sites`);

    if (sitesResp.ok) {

      const sites = await sitesResp.json();

      state.sites = sites.reduce((a, s) => { a[s.id] = s.name; return a; }, {});

    }


    await loadAPs();

    showDashboard();

  } catch (e) {

    showError(e.message || 'Connection failed. Check credentials and try again.');

    btn.disabled = false; btn.textContent = 'Connect & Load APs';

  }

}


async function loadAPs() {

  const { orgId } = state;


  // Fetch devices (APs) — paginate if needed

  let allAPs = [];

  let page = 1;

  while (true) {

    const resp = await mistFetch(`/api/v1/orgs/${orgId}/devices/search?type=ap&limit=200&page=${page}`);

    if (!resp.ok) throw new Error(`Failed to load APs (${resp.status}).`);

    const data = await resp.json();

    const results = data.results || [];

    allAPs = allAPs.concat(results);

    if (results.length < 200) break;

    page++;

  }


  // Fetch client counts for all sites

  const clientMap = {};

  try {

    const statsResp = await mistFetch(`/api/v1/orgs/${orgId}/stats/devices?type=ap&limit=1000`);

    if (statsResp.ok) {

      const statsData = await statsResp.json();

      const stats = statsData.results || statsData || [];

      stats.forEach(s => {

        if (s.mac) clientMap[s.mac] = s.num_clients || 0;

      });

    }

  } catch (_) {}


  state.aps = allAPs.map(ap => ({

    id: ap.id || ap.device_id,

    name: ap.name || ap.hostname || ap.mac,

    mac: ap.mac,

    model: ap.model || '—',

    site_id: ap.site_id,

    site_name: state.sites[ap.site_id] || ap.site_id || '—',

    status: ap.status === 'connected' ? 'connected' : 'disconnected',

    ip: ap.ip_address || ap.ip || '—',

    clients: clientMap[ap.mac] || ap.num_clients || 0,

  }));


  updateStats();

  renderTable();

  setStatus('live', `Live · ${new Date().toLocaleTimeString()}`);

  document.getElementById('last-updated').textContent = `Last updated: ${new Date().toLocaleTimeString()}`;

}


// ── API HELPER ─────────────────────────────────────────────────

function mistFetch(path) {

  return fetch(state.baseUrl + path, {

    headers: { Authorization: `Token ${state.token}`, 'Content-Type': 'application/json' }

  });

}


// ── STATS ──────────────────────────────────────────────────────

function updateStats() {

  const { aps, orgId } = state;

  const connected = aps.filter(a => a.status === 'connected');

  const totalClients = aps.reduce((s, a) => s + a.clients, 0);

  const busiest = [...aps].sort((a,b) => b.clients - a.clients)[0];


  document.getElementById('stat-total').textContent = aps.length;

  document.getElementById('stat-org').textContent = orgId.slice(0,8) + '…';

  document.getElementById('stat-connected').textContent = connected.length;

  document.getElementById('stat-connected-pct').textContent =

    aps.length ? Math.round(connected.length / aps.length * 100) + '% online' : '—';

  document.getElementById('stat-clients').textContent = totalClients;

  if (busiest) {

    document.getElementById('stat-busiest-count').textContent = busiest.clients;

    document.getElementById('stat-busiest-name').textContent = busiest.name;

  }

}


// ── RENDER TABLE ───────────────────────────────────────────────

function renderTable() {

  const { aps, sort, filter } = state;

  const tbody = document.getElementById('ap-tbody');


  let list = [...aps];


  // filter

  if (filter) {

    const q = filter.toLowerCase();

    list = list.filter(a =>

      a.name.toLowerCase().includes(q) ||

      a.mac.toLowerCase().includes(q) ||

      a.site_name.toLowerCase().includes(q) ||

      a.model.toLowerCase().includes(q)

    );

  }


  // sort

  if (sort === 'clients') list.sort((a,b) => b.clients - a.clients);

  else if (sort === 'name') list.sort((a,b) => a.name.localeCompare(b.name));

  else if (sort === 'status') list.sort((a,b) => (a.status === 'connected' ? -1 : 1));


  if (!list.length) {

    tbody.innerHTML = `<tr><td colspan="6"><div class="state-block">

      <div class="icon">⊘</div>

      <h3>No APs found</h3>

      <p>${filter ? 'No results match your filter.' : 'No access points returned from the API.'}</p>

    </div></td></tr>`;

    return;

  }


  const maxClients = Math.max(...list.map(a => a.clients), 1);


  tbody.innerHTML = list.map(ap => {

    const pct = Math.round((ap.clients / maxClients) * 100);

    const clientColor = ap.clients === 0 ? 'var(--muted)' :

                        ap.clients < 10 ? 'var(--accent)' :

                        ap.clients < 25 ? 'var(--amber)' : 'var(--red)';

    return `

    <tr>

      <td>

        <div class="ap-name">${esc(ap.name)}</div>

        <div class="ap-mac">${esc(ap.mac)}</div>

      </td>

      <td class="hide-mobile"><span class="site-pill">${esc(ap.site_name)}</span></td>

      <td class="hide-mobile"><span class="model-text">${esc(ap.model)}</span></td>

      <td>

        <span class="badge ${ap.status}">

          <span class="dot" style="width:5px;height:5px;border-radius:50%;background:currentColor;display:inline-block"></span>

          ${ap.status}

        </span>

      </td>

      <td>

        <div class="client-bar-wrap">

          <span class="client-count" style="color:${clientColor}">${ap.clients}</span>

          <div class="bar-track">

            <div class="bar-fill" style="width:${pct}%"></div>

          </div>

        </div>

      </td>

      <td class="hide-mobile"><span class="ip-text">${esc(ap.ip)}</span></td>

    </tr>`;

  }).join('');

}

// ── LOGIN ──────────────────────────────────────────────────────

// To add/change users: edit the USERS object below.

// Keys are usernames, values are SHA-256 hashes of passwords.

// Generate a hash at: https://emn178.github.io/online-tools/sha256.html

// Default: admin / mist2024

const USERS = {

  "admin": "b1c978caec9c6cc91dfdf66e94887cb63e24d74a898cd44413a76ee35945e4f",

  // "viewer": "your-sha256-hash-here",

};


async function sha256(str) {

  const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(str));

  return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2,'0')).join('');

}


async function doLogin() {

  const user = document.getElementById('login-user').value.trim();

  const pass = document.getElementById('login-pass').value;

  const errEl = document.getElementById('login-error');

  errEl.textContent = '';


  if (!user || !pass) {

    setLoginError('Please enter both username and password.');

    return;

  }


  const hash = await sha256(pass);

  if (USERS[user] && USERS[user] === hash) {

    sessionStorage.setItem('mist_auth', '1');

    const overlay = document.getElementById('login-overlay');

    overlay.classList.add('out');

    setTimeout(() => { overlay.style.display = 'none'; }, 350);

    document.getElementById('logout-btn').style.display = 'inline-block';

  } else {

    setLoginError('Invalid username or password.');

    ['login-user','login-pass'].forEach(id => {

      const el = document.getElementById(id);

      el.classList.add('shake');

      el.addEventListener('animationend', () => el.classList.remove('shake'), { once: true });

    });

  }

}
